const config = { type: Phaser.AUTO, width: window.innerWidth, height: window.innerHeight, parent: "game-container", backgroundColor: "#000000", scene: { preload: preload, create: create } }; new Phaser.Game(config); function preload() { this.load.image('map', 'assets/map.png?v=' + Date.now()); this.load.image('goose', 'assets/goose.png'); this.load.image('dog', 'assets/dog.png'); } function create() { const map = this.add.image( window.innerWidth / 2, window.innerHeight / 2, 'map' ); const scale = Math.min( window.innerWidth / map.width, window.innerHeight / map.height ); map.setScale(scale); const goose = this.add.image(300, 300, 'goose'); goose.setScale(0.5); const dog = this.add.image(600, 300, 'dog'); dog.setScale(0.5); this.add.text(20, 20, "CRITTER CRIMES LIVE", { fontSize: "20px", color: "#ffffff" }); }
